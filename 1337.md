## files
```c++
#include <iostream>
#include <fstream>

using namespace std;

int main() {
    string fileName;
    cout << "Enter file name: ";
    cin >> fileName;


    // Writing to a file using ofstream
    ofstream outputFile;
    outputFile.open(fileName);
    if (!outputFile) {
        cout << "Could not open file for writing\nExiting\n";
        return 1;
    }
    outputFile << "Hello, this is a sample text file.\n";
    outputFile << "Learning file handling in C++.\n";
    outputFile.close(); // Always close files after use





    // Reading from a file using ifstream line by line
    ifstream inputFile;
    inputFile.open(fileName);
    if (!inputFile) {
        cout << "Could not open file for reading\nExiting\n";
        return 1;
    }
    cout << "\nContents of the file:\n";
    string line;
    while (getline(inputFile, line)) {
        cout << line << endl;
    }
    inputFile.close();
    
    
    
    
    // Reading from a file using ifstream word by word
    inputFile.open(fileName);
    if (!inputFile) {
        cout << "Could not open file for reading.\n";
        return 1;
    }
    cout << "\nContents of the file (word by word):\n"; 
    string word;
    while (inputFile >> word) { // Reads one word at a time
        cout << word << endl;
    }
    inputFile.close();




    // Appending to a file using fstream
    fstream file;
    file.open(fileName, ios::app);
    if (!file) {
        cout << "Could not open file for appending\nExiting\n";
        return 1;
    }
    file << "Appending a new line to the file.\n";
    file.close();

    cout << "\nFile updated successfully!\n";
    return 0;
}
```

## linear search
```c++
#include <iostream>

using namespace std;

//prototype
int linearSearch(int arr[], int size, int value);

// function main begins program execution
int main()
{
    int arr[] = { 2, 4, 7, 10, 11, 28, 34, 45, 50, 59, 60, 66, 69, 70, 75, 89, 95 };
    int size = sizeof(arr)/sizeof(int), value;

    cout <<"Enter the element to search: ";
    cin >> value;
    
    int result = linearSearch(arr, size, value);

    if (result != -1)
    {
        cout << value << " is at index " << result << endl;
    }
    else
    {
        cout << value << " is not found!" << endl;
    }
    return 0;
}
int linearSearch(int arr[], int size, int value)
{   
    int index = 0;      // Used as a subscript to search the array  
    int position = -1;  // To record the position of search value   
    bool found = false; // Flag to indicate if value was found   
    while (index < size && !found)   
    {      
        if (arr[index] == value) // If the value is found       
        {          
            found = true; // Set the flag          
            position = index; // Record the value's subscript      
        }      
        index++; // Go to the next element   
    }
    return position; // Return the position, or -1
}

```
## binary search
```c++
#include <iostream>

using namespace std;

//prototype
int binarySearch(int array[], int size, int value);

// function main begins program execution
int main()
{
    int array[] = { 2, 4, 7, 10, 11, 28, 34, 45, 50, 59, 60, 66, 69, 70, 75, 89, 95 };
    int value, size = 17;
    cout <<"Enter the element to search: ";
    cin >> value;
    int result = binarySearch(array, size, value);
    if (result != -1)
    {
        cout << value << " is at index " << result << endl;
    }
    else
    {
        cout << value << " is not found!" << endl;
    }

    return 0;

}

int binarySearch(int array[], int size, int value)
{   
    int first = 0,             // First array element       
    last = size - 1,       // Last array element       
    middle,                // Mid point of search       
    position = -1;         // Position of search value   
    bool found = false;        // Flag   
    while (!found && first <= last)   
    {      
        middle = (first + last) / 2;     // Calculate mid point      
        if (array[middle] == value)      // If value is found at mid      
        {         
            found = true;         
            position = middle;      
        }else if (array[middle] > value)  // If value is in lower half   
        {
            last = middle - 1;      
        }else{         
            first = middle + 1; // If value is in upper half   
        }
    }
    return position;
}
```

## bubble sort
```c++

#include <iostream>

using namespace std;

//prototype
void bubbleSort(int arr[], int size);
void printArray(int arr[], int size);

int main()
{
    int arr[] = {  50, 59, 60, 66, 69, 10, 11, 28, 34, 2, 4, 7, 45, 70, 75, 89, 95 };
    int size = sizeof(arr)/sizeof(int);

    cout << "Before:" << endl;
    printArray(arr,size);
    
    bubbleSort(arr, size);
    
    cout << "After:" << endl;
    printArray(arr,size);

    return 0;
}
void bubbleSort(int arr[], int size) {
    // we are comparing the jth element with j+1th element
    for (int i = 0; i < size - 1; i++) { // Number of passes: size - 1 because the first element is already the smallest if the last n-1 elements are the largest
        bool swapped = false;
        for (int j = 0; j < size - i - 1; j++) { // Comparing adjacent elements: upto size - 1 is because we are comparing with the next element so we should not reach last element
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]); // Swap if elements are out of order
                swapped = true;
            }
        }
        if (!swapped) break; // If no swaps in a pass, the array is sorted
    }
}
void printArray(int arr[], int size){
    for(int i=0;i<size;i++){
        cout << arr[i] << " ";
    }
    cout << endl;
}

```

## selection sort
```c++
#include <iostream>

using namespace std;

//prototype
void selectionSort(int arr[], int size);
void printArray(int arr[], int size);

int main()
{
    int arr[] = {  50, 59, 60, 66, 69, 10, 11, 28, 34, 2, 4, 7, 45, 70, 75, 89, 95 };
    int size = sizeof(arr)/sizeof(int);

    cout << "Before:" << endl;
    printArray(arr,size);
    
    selectionSort(arr, size);
    
    cout << "After:" << endl;
    printArray(arr,size);

    return 0;
}
void selectionSort(int array[], int size) {
    int minIndex, minValue;

    for (int start = 0; start < size - 1; start++) {
        minIndex = start;
        minValue = array[start];

        for (int index = start + 1; index < size; index++) {
            if (array[index] < minValue) {
                minValue = array[index];
                minIndex = index;
            }
        }
        // Swap the found minimum element with the first element of the unsorted part
        swap(array[minIndex], array[start]);
    }
}

void printArray(int arr[], int size){
    for(int i=0;i<size;i++){
        cout << arr[i] << " ";
    }
    cout << endl;
}
```

## insertion sort
```c++
#include <iostream>

using namespace std;

//prototype
void insertionSort(int arr[], int size);
void printArray(int arr[], int size);

int main()
{
    int arr[] = {  50, 59, 60, 66, 69, 10, 11, 28, 34, 2, 4, 7, 45, 70, 75, 89, 95 };
    int size = sizeof(arr)/sizeof(int);
    
    cout << "Before:" << endl;
    printArray(arr,size);
    
    insertionSort(arr, size);
    
    cout << "After:" << endl;
    printArray(arr,size);

    return 0;
}
void insertionSort(int arr[], int size) {
    // Iterate over the array starting from the second element
    for (int currentIndex = 1; currentIndex < size; currentIndex++) {
        int currentValue = arr[currentIndex]; // The element we want to insert correctly
        int previousIndex = currentIndex - 1; // Start checking from the previous index

        // Shift elements to the right until we find the correct position for currentValue
        while (previousIndex >= 0 && arr[previousIndex] > currentValue) {
            arr[previousIndex + 1] = arr[previousIndex]; // Shift element to the right
            previousIndex--; // Move left to continue checking
        }

        // Place the currentValue in its correct position
        arr[previousIndex + 1] = currentValue;
    }
}

void printArray(int arr[], int size){
    for(int i=0;i<size;i++){
        cout << arr[i] << " ";
    }
    cout << endl;
}
```
