## Inheritance Example: PerishableProduct inherits from Product
```c++
// PerishableProduct.h
#ifndef PERISHABLEPRODUCT_H
#define PERISHABLEPRODUCT_H

#include "Product.h"

class PerishableProduct : public Product { // public inheritance
private:
    string expirationDate;

public:
    PerishableProduct(); // Default constructor
    PerishableProduct(string p, string n, int t, double pr, int inv, string expDate);

    void setExpirationDate(string date);
    string getExpirationDate() const;
};

#endif
```

```c++
// PerishableProduct.cpp
#include "PerishableProduct.h"

// Default constructor
PerishableProduct::PerishableProduct() : Product() {
    expirationDate = "N/A";
}

// Parameterized constructor using base class constructor
PerishableProduct::PerishableProduct(string p, string n, int t, double pr, int inv, string expDate)
    : Product(p, n, t, pr, inv) {
    expirationDate = expDate;
}

void PerishableProduct::setExpirationDate(string date) {
    expirationDate = date;
}

string PerishableProduct::getExpirationDate() const {
    return expirationDate;
}

```

## Sample main.cpp with Class & Inheritance Usage
```c++
#include <iostream>
#include "Product.h"
#include "PerishableProduct.h"

using namespace std;

int main() {
    // Using parameterized constructor
    Product p1("123", "Apple", 1, 0.99, 100);
    
    // Using setters
    Product p2;
    p2.setProductAttributes("456", "Banana", 1, 0.59, 50);

    // Display using getters
    cout << "Product: " << p1.getName() << ", Price: $" << p1.getPrice() << endl;

    // Inheritance: Perishable product
    PerishableProduct milk("789", "Milk", 2, 3.49, 30, "2025-05-01");
    cout << "Perishable Product: " << milk.getName() << ", Expiry: " << milk.getExpirationDate() << endl;

    return 0;
}

```




## Pointers and classes
```c++

#include "Product.h"
#include "PerishableProduct.h"

int main() {
    // pointer method 1
    Product *prodPointer = nullptr;
    prodPointer = new Product;
    prodPointer -> setProductAttributes("456", "Banana", 1, 0.59, 50);
    
    // pointer method 2
    Product* prodPointer = nullptr;
    Product p1;
    prodPointer = &p1;

    // Set using pointer
    prodPointer->setProductAttributes("456", "Banana", 1, 0.59, 50);
    // Get using pointer
    cout << "Product: " << prodPointer->getName() << ", Price: $" << prodPointer->getPrice() << endl;

    // for pointer method 1
    delete prodPointer;
    prodPointer = nullptr;

    return 0;
}
```
